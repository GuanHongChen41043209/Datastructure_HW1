## 宣告根節點的結構
```cpp
struct TreeNode{
    int key;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int k){
        key = k;
        left = nullptr;
        right = nullptr;
    }
};
```
## 把值新增到二元樹
```cpp
TreeNode* Insert(TreeNode* root, int x, bool& inserted){
    if(root == nullptr){
        inserted = true;
        return new TreeNode(x);
    }
    if(x < root->key){
        root->left = Insert(root->left, x, inserted);
    }
    else if(x > root->key){
        root->right = Insert(root->right, x, inserted);
    }
    else{
        inserted = false;
    }
    return root;
}
```
## 回傳二元樹的高度
```cpp
int Height(TreeNode* root){
    if(root == nullptr){
        return 0;
    }
    int leftHeight = Height(root->left);
    int rightHeight = Height(root->right);
    if(leftHeight > rightHeight){
        return 1 + leftHeight;
    }
    else{
        return 1 + rightHeight;
    }
}
```

## 回傳二元樹中最小的值
```cpp
TreeNode* FindMin(TreeNode* root, int& steps){
    if(root == nullptr){
        return nullptr;
    }
    while(root->left != nullptr){
        steps++;
        root = root->left;
    }
    return root;
}
```
## 刪除二元樹中指定的k值
```cpp
TreeNode* Delete(TreeNode* root, int k, int& steps){
    steps++;
    if(root == nullptr){
        return nullptr;
    }
    if(k < root->key){
        root->left = Delete(root->left, k, steps);
    }
    else if(k > root->key){
        root->right = Delete(root->right, k, steps);
    }
    else{
        if(root->left == nullptr && root->right == nullptr){
            delete root;
            return nullptr;
        }
        else if(root->left == nullptr){
            TreeNode* temp = root->right;
            delete root;
            return temp;
        }
        else if(root->right == nullptr){
            TreeNode* temp = root->left;
            delete root;
            return temp;
        }
        else{
            TreeNode* temp = FindMin(root->right, steps);
            root->key = temp->key;
            root->right = Delete(root->right, temp->key, steps);
        }
    }
    return root;
}
```
## 尋找二元樹中指定的k值

bool Search(TreeNode* root, int k){
    if(root == nullptr){
        return false;
    }
    if(k == root->key){
        return true;
    }
    else if(k < root->key){
