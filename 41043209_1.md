## 宣告最小優先佇列的類別
```cpp
template <class T>
class MinPQ{
    public:
    virtual ~MinPQ(){}
    virtual bool IsEmpty() const = 0;
    virtual const T& Top() const = 0;
    virtual void Push(const T&) = 0;
    virtual void Pop() = 0;
};
```
## 宣告最小堆疊的類別
```cpp
template <class T>
class MinHeap : public MinPQ<T>{
    private:
    T* heap;
    int capacity;
    int size;
    public:
    MinHeap(int theCapacity = 10);
    ~MinHeap();

    bool IsEmpty() const override;
    const T& Top() const override;
    void Push(const T& x) override;
    void Pop() override;
};
```
## 定義MinHeap的建構子
```cpp
template <class T>
MinHeap<T>::MinHeap(int theCapacity){
    capacity = theCapacity;
    size = 0;
    heap = new T[capacity + 1];
}
```
## 定義MinHeap的解構子
```cpp
template <class T>
MinHeap<T>::~MinHeap(){
    delete[] heap;
}
```
## 判斷MinHeap是否為空
```cpp
template <class T>
bool MinHeap<T>::IsEmpty() const{
    return size == 0;
}
```
## 回傳heap的第一個值
```cpp
template <class T>
const T& MinHeap<T>::Top() const{
    if(IsEmpty()){
        cout << "Heap is empty!" << endl;
        exit(1);
    }
    return heap[1];
}
```
## 新增值到MinHeap
```cpp
template <class T>
void MinHeap<T>::Push(const T& x){
    if(size == capacity){
        cout << "Heap is full!" << endl;
        return;
    }
    int i = ++size;
    while(i != 1 && x < heap[i / 2]){
        heap[i] = heap[i / 2];
        i /= 2;
    }
    heap[i] = x;
}
```
## 移除MinHeap的第一的值
```cpp
template <class T>
void MinHeap<T>::Pop(){
    if(IsEmpty()){
        cerr << "Heap is empty!" << endl;
        return;
    }
    T lastElement = heap[size--];
    int current = 1;
    int child = 2;
    while(child <= size){
        if(child < size && heap[child + 1] < heap[child]){
            child++;
        }
        if(lastElement <= heap[child]){
            break;
        }
        heap[current] = heap[child];
        current = child;
        child = current * 2;
    }
    heap[current] = lastElement;
}
```
